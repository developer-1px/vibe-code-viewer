body.statements.forEach((stmt) => {
    // 1. Look for comments before this statement
    const comments = ts.getLeadingCommentRanges(sourceFile.text, stmt.getFullStart());

    if (comments && comments.length > 0) {
      comments.forEach(comment => {
        let text = sourceFile.text.substring(comment.pos, comment.end);
        const isBlockComment = text.startsWith('/*');
        text = text.replace(/^\/\/\s*/, '').replace(/^\/\*\*\s*/, '').replace(/^\/\*\s*/, '').replace(/\*\/$/, '').replace(/^\s*\*\s?/gm, '').trim();

        if (!text) return;

        // [branch] Special Tags
        // [NOTE], [branch] 같은 특수 태그를 감지하여 Prose 블록이 아닌 별도 타입으로 분류합니다.
        const tagMatch = text.match(/^\[(\w+)\]\s*(.*)/s);
        if (tagMatch) {
          const [, label, content] = tagMatch;
          const type = label.toLowerCase() === 'branch' ? BlockType.BRANCH : BlockType.TAG;
          blocks.push({ type, label, content, lines: `L${sourceFile.getLineAndCharacterOfPosition(comment.pos).line + 1}` });
        } else {
          blocks.push({ type: BlockType.PROSE, content: text });
        }
      });
    }

    blocks.push({
        type: BlockType.CODE,
        content: stmt.getText(sourceFile),
        lines: getLineRange(sourceFile, stmt)
    });

    lastPos = stmt.getEnd();
  });

return blocks;

ts.forEachChild(sourceFile, node => {
    // --- IMPORTS ---
    if (ts.isImportDeclaration(node)) {
        const path = (node.moduleSpecifier as ts.StringLiteral).text;
        if (node.importClause && node.importClause.namedBindings && ts.isNamedImports(node.importClause.namedBindings)) {
            node.importClause.namedBindings.elements.forEach(el => {
                imports.push({ name: el.name.text, path });
            });
        }
    }

    // --- FUNCTIONS ---
    else if (ts.isFunctionDeclaration(node)) {
        const name = node.name?.text || 'anonymous';
        const isExport = node.modifiers?.some(m => m.kind === ts.SyntaxKind.ExportKeyword);
        if (isExport) exports.push({ name, type: 'function' });

        const { description, tags } = parseJSDoc(node);
        const signatureEnd = node.body ? node.body.getStart() : node.end;
        const signature = sourceFile.text.substring(node.getStart(), signatureEnd).trim();

        symbols.push({
            name,
            type: 'function',
            modifiers: node.modifiers?.map(m => m.getText(sourceFile)) || [],
            lineRange: getLineRange(sourceFile, node),
            signature,
            description,
            parameters: node.parameters.map(p => ({
                name: p.name.getText(sourceFile),
                type: p.type?.getText(sourceFile) || 'any',
                description: tags[p.name.getText(sourceFile)] || ''
            })),
            returns: node.type?.getText(sourceFile) || 'void',
            throws: [],
            analysis: {
                complexity: calculateComplexity(node),
                branches: 0,
                coverage: 'N/A'
            },
            blocks: node.body ? generateBlocks(sourceFile, node.body) : []
        });
    }
  });
